# -*- mode: cmake; -*-
set (hdrs
  assert.hpp
  pc.hpp
  util.hpp
  )

  pc.cpp
  )

ADD_LIBRARY(fvm-pc.shared SHARED ${srcs})
set_target_properties(fvm-pc.shared PROPERTIES OUTPUT_NAME fvm-pc)

  ARCHIVE
  DESTINATION libexec
  COMPONENT modules)
install(TARGETS fvm-pc.shared
  LIBRARY
  DESTINATION libexec
  COMPONENT modules)
